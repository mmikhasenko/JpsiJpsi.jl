  H_{\lambda_1,\lambda_2} = (-1)^{j_1-\lambda_1} \sum_{ls} \sqrt{\frac{2l+1}{2j+1}}

  \left\langle L,0;s,\lambda_1-\lambda_2|j,\lambda_1-\lambda_2 \right\rangle H_{ls}.
\end{align}
